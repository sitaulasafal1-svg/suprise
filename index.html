// PREVIEW OPEN function openPreview(src) { const preview = document.getElementById("preview"); const img = document.getElementById("previewImg"); img.src = src; preview.classList.add("show"); } // CLOSE PREVIEW document.getElementById("preview").onclick = function() { this.classList.remove("show"); }; // DRAG + CLICK // DRAG + CLICK document.querySelectorAll(".photo").forEach(photo => { let isDragging = false; let offsetX = 0; let offsetY = 0; // TOUCH START photo.addEventListener("touchstart", e => { const touch = e.touches[0]; const rect = photo.getBoundingClientRect(); offsetX = touch.clientX - rect.left; offsetY = touch.clientY - rect.top; isDragging = false; }); // TOUCH MOVE photo.addEventListener("touchmove", e => { const touch = e.touches[0]; isDragging = true; photo.style.left = (touch.clientX - offsetX) + "px"; photo.style.top = (touch.clientY - offsetY) + "px"; photo.style.transform = "translate(0, 0) scale(1)"; }); // TOUCH END photo.addEventListener("touchend", () => { if (!isDragging) { openPreview(photo.src); } }); // MOUSE SUPPORT photo.addEventListener("mousedown", e => { const rect = photo.getBoundingClientRect(); offsetX = e.clientX - rect.left; offsetY = e.clientY - rect.top; isDragging = false; function move(e) { isDragging = true; photo.style.left = (e.clientX - offsetX) + "px"; photo.style.top = (e.clientY - offsetY) + "px"; photo.style.transform = "translate(0,0) scale(1)"; } function up() { document.removeEventListener("mousemove", move); document.removeEventListener("mouseup", up); if (!isDragging) { openPreview(photo.src); } } document.addEventListener("mousemove", move); document.addEventListener("mouseup", up); }); });
